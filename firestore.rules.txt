rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ========== 공통 함수 ==========
    function isLoggedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return isLoggedIn() && request.auth.uid == uid;
    }

    // ⭐ 관리자 UID — 본인 Firebase UID로 교체하세요
    function isAdmin() {
      return isLoggedIn() && request.auth.uid in [
        'LIuwCkomQmf0iHf7QoBXs57OecI2'
      ];
    }


    // ========== products 컬렉션 ==========
    match /products/{productId} {
      allow read: if true;

      allow create: if isLoggedIn()
        && request.resource.data.sellerId == request.auth.uid
        && request.resource.data.title is string
        && request.resource.data.title.size() > 0
        && request.resource.data.title.size() <= 50
        && request.resource.data.price is number
        && request.resource.data.price >= 0;

      allow update: if isLoggedIn() && (
        isAdmin()
        || resource.data.sellerId == request.auth.uid
        || request.resource.data.diff(resource.data).affectedKeys()
            .hasOnly(['views', 'likeCount', 'chatCount', 'likedBy'])
      );

      allow delete: if isLoggedIn()
        && (resource.data.sellerId == request.auth.uid || isAdmin());
    }


    // ========== users 컬렉션 ==========
    match /users/{userId} {
      allow read: if isLoggedIn();
      allow create: if isOwner(userId);
      allow update: if isOwner(userId) || isAdmin();
      allow delete: if isOwner(userId);
    }


    // ========== chats 컬렉션 ==========
    match /chats/{chatId} {

      // ✅ FIX: 관리자도 읽기 가능 (admin.html 채팅 모니터링용)
      allow read: if isLoggedIn() && (
        resource.data.buyerId  == request.auth.uid ||
        resource.data.sellerId == request.auth.uid ||
        isAdmin()
      );

      allow create: if isLoggedIn()
        && request.resource.data.buyerId == request.auth.uid;

      // ✅ FIX: 관리자도 업데이트 가능
      allow update: if isLoggedIn() && (
        resource.data.buyerId  == request.auth.uid ||
        resource.data.sellerId == request.auth.uid ||
        isAdmin()
      );

      allow delete: if false;

      match /messages/{messageId} {
        allow read, create: if isLoggedIn() && (
          get(/databases/$(database)/documents/chats/$(chatId)).data.buyerId  == request.auth.uid ||
          get(/databases/$(database)/documents/chats/$(chatId)).data.sellerId == request.auth.uid
        );
        allow update: if false;
        allow delete:  if false;
      }
    }


    // ========== reports 컬렉션 ==========
    // ✅ FIX: 기존에 규칙 자체가 없어서 admin.html 신고 관리 전체 불동작했던 원인
    match /reports/{reportId} {
      // 읽기·수정: 관리자만
      allow read:   if isAdmin();
      allow update: if isAdmin();
      allow delete: if false;

      // 신고 생성: 로그인한 사용자 누구나 (본인 uid 포함 필수)
      allow create: if isLoggedIn()
        && request.resource.data.reporterId == request.auth.uid
        && request.resource.data.type   is string
        && request.resource.data.reason is string;
    }


    // ========== inquiries 컬렉션 ==========
    match /inquiries/{inquiryId} {
      allow read:   if isAdmin();
      allow update: if isAdmin();
      allow delete: if false;

      allow create: if
        request.resource.data.name    is string && request.resource.data.name.size()    > 0 &&
        request.resource.data.email   is string && request.resource.data.email.size()   > 0 &&
        request.resource.data.type    is string && request.resource.data.type.size()    > 0 &&
        request.resource.data.title   is string && request.resource.data.title.size()   > 0 &&
        request.resource.data.content is string && request.resource.data.content.size() > 0 &&
        request.resource.data.content.size() <= 1000;
    }

  }
}
